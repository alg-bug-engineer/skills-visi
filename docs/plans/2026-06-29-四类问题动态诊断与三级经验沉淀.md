```python
# backend/intersection_agent/services/dimension_pack_service.py
from __future__ import annotations
from pathlib import Path
import yaml
from intersection_agent.config import get_settings

class DimensionPackService:
    def __init__(self) -> None:
        path = Path(__file__).resolve().parent.parent / "config" / "problem_dimension_packs.yaml"
        with open(path, encoding="utf-8") as f:
            self._cfg = yaml.safe_load(f)

    def focus_categories(self, problem_types: list[str]) -> list[str]:
        cats: list[str] = list(self._cfg.get("base", {}).get("focus_categories", []))
        packs = self._cfg.get("packs", {})
        for pt in problem_types:
            for c in packs.get(pt, {}).get("focus_categories", []):
                if c not in cats:
                    cats.append(c)
        return cats
```

> 注：`channelization` 类目当前规则未标 `focus_category`。本任务后置 Task A1b：在 `traffic_rules.yaml` 给 `rule_channelization_conflict` 补 `focus_category: channelization`，给相序冲突类规则补标。运行 `pytest tests/test_rule_engine.py -v` 确认未回归。
