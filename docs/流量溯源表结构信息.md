-- ============================================================
-- 表名：dws_tfc_inter_turn_flow_correlate_m
-- 业务：路口转向流量上下游关系统计表（月度聚合）
-- 层级：DWS 汇总主题资产层 — 交通运行主题库
-- 说明：以路口进口道转向为基准，按月统计其车流与上/下游
--       关联路口转向之间的流量比例关系。支撑交通智能体
--       截流策略分析、信号协调优化和拥堵传播路径诊断。
-- 主键：month + inter_id + f_dir8_no + turn_dir_no
--       + cor_inter_id + cor_f_dir8_no + cor_turn_dir_no
--       + trace_type
-- 更新周期：月更新
-- ============================================================
CREATE TABLE IF NOT EXISTS `dws_tfc_inter_turn_flow_correlate_m` (
    -- 维度字段
    `month`              VARCHAR(6)   NOT NULL COMMENT '统计月份，格式 yyyyMM',
    `day_of_week`        VARCHAR(10)  NOT NULL COMMENT '星期类型：周一/周二/.../周日，工作日/非工作日',
    `period_type`        VARCHAR(10)  NOT NULL COMMENT '时段类型：MORNING_PEAK-早高峰，EVENING_PEAK-晚高峰，OFF_PEAK-平峰',
    `inter_id`           VARCHAR(64)  NOT NULL COMMENT '基准路口16位编码',
    `f_dir8_no`          TINYINT      NOT NULL COMMENT '基准路口进口道8方向编码：0-北向南(北进口)，1-东北向西南(东北进口)，2-东向西(东进口)，3-东南向西北(东南进口)，4-南向北(南进口)，5-西南向东北(西南进口)，6-西向东(西进口)，7-西北向东南(西北进口)',
    `turn_dir_no`        TINYINT      NOT NULL COMMENT '基准路口车流转向类型：1-左转，2-直行，3-右转，4-掉头',
    `cor_inter_id`       VARCHAR(64)  NOT NULL COMMENT '关联路口16位编码',
    `cor_f_dir8_no`      TINYINT      NOT NULL COMMENT '关联路口进口道8方向编码（同f_dir8_no字典）',
    `cor_turn_dir_no`    TINYINT      NOT NULL COMMENT '关联路口车流转向类型：1-左转，2-直行，3-右转，4-掉头',

    -- 业务指标字段
    `flow_share_ratio`   DECIMAL(5,2) NOT NULL COMMENT '车流占比(%)：关联路口转向的车流占本路口(基准路口)对应转向车流的比例。如值为15.50，表示本路口该转向的流量中有15.50%来自/去往该关联路口转向',
    `trace_type`         VARCHAR(8)   NOT NULL COMMENT '溯源类型：UPSTREAM-上游来源(来自关联路口)，DOWNSTREAM-下游去向(去往关联路口)',

    -- 审计与管理字段
    `create_time`        DATETIME     NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '记录创建时间',
    `update_time`        DATETIME     NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '记录更新时间',
    `is_deleted`         TINYINT      NOT NULL DEFAULT 0 COMMENT '逻辑删除标记：0-有效，1-已删除',

    -- 主键约束
    PRIMARY KEY (`month`, `inter_id`, `f_dir8_no`, `turn_dir_no`, `cor_inter_id`, `cor_f_dir8_no`, `cor_turn_dir_no`, `trace_type`)

) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
  COMMENT='路口转向流量上下游关系统计表(月度聚合)DWS层-交通运行主题库';

-- ============================================================
-- 索引设计
-- ============================================================

-- 索引1：按基准路口+溯源类型查询，用于智能体截流分析
ALTER TABLE `dws_tfc_inter_turn_flow_correlate_m`
    ADD INDEX `idx_inter_trace` (`inter_id`, `trace_type`, `month`);

-- 索引2：按关联路口+溯源类型查询，用于逆向溯源分析
ALTER TABLE `dws_tfc_inter_turn_flow_correlate_m`
    ADD INDEX `idx_cor_inter_trace` (`cor_inter_id`, `trace_type`, `month`);

-- 索引3：按月+时段类型+星期类型查询，用于典型日分析
ALTER TABLE `dws_tfc_inter_turn_flow_correlate_m`
    ADD INDEX `idx_month_period_week` (`month`, `period_type`, `day_of_week`);

-- 索引4：按基准路口8方向编码查询，用于渠化诊断
ALTER TABLE `dws_tfc_inter_turn_flow_correlate_m`
    ADD INDEX `idx_inter_dir8` (`inter_id`, `f_dir8_no`, `month`);

-- 字段说明（备注）
-- ============================================================
-- 1. f_dir8_no / cor_f_dir8_no 采用 TINYINT 类型存储8方向枚举值，
--    符合数仓规范中编码字段用数值类型存储的标准。
--    字典映射（从北向南为0，顺时针递增）：
--      0 = 由北向南（北进口，南行方向）
--      1 = 由东北向西南（东北进口，西南行方向）
--      2 = 由东向西（东进口，西行方向）
--      3 = 由东南向西北（东南进口，西北行方向）
--      4 = 由南向北（南进口，北行方向）
--      5 = 由西南向东北（西南进口，东北行方向）
--      6 = 由西向东（西进口，东行方向）
--      7 = 由西北向东南（西北进口，东南行方向）
--
-- 2. turn_dir_no 转向编码：
--      1 = 左转
--      2 = 直行
--      3 = 右转
--      4 = 掉头
--
-- 3. trace_type 溯源类型：
--      'UPSTREAM'   = 上游来源：基准路口的车流来自关联路口转向
--      'DOWNSTREAM' = 下游去向：基准路口的车流流向关联路口转向
--
-- 4. flow_share_ratio 分母为基准路口对应转向（即 inter_id + f_dir8_no +
--    turn_dir_no 组合）的车流总量，分子为来自/去往该关联路口转向
--    （即 cor_inter_id + cor_f_dir8_no + cor_turn_dir_no 组合）的车流量。
--    本路口同一转向的多个关联路口占比之和理论上不超过100。
--
 ============================================================

## 应用层：拓扑一跳 vs 统计途经（2026-06-30）

地图「单链路发光溯源」**不得**仅用 `flow_correlate` 最大 `flow_share_ratio` 定上一路口（会选到地理方位错误的路口，如西进口误跳东侧）。

| 层级 | 数据源 | 用途 |
|------|--------|------|
| **拓扑一跳（主）** | `road6.dim_link_info.geom` + `dwd_tfc_rltn_wide_inter_ft_link` | 确定地理相邻上一路口、折线路径 |
| **佐证** | `xianchang.dws_tfc_inter_turn_flow_correlate_m` | 在白名单邻接路口内取流量占比，展示「流量占比 xx%」 |
| **干线序（可选）** | `dim_line_inter_rltn` | 校验主线上一跳，与 link 邻接交叉验证 |

实现：`UpstreamTopologyService.pick_upstream_hop`；嗅探脚本：`backend/scripts/probe_upstream_topology.py`。
